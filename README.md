  # Install and run API
  
  ```
# Clone this repository
$ git clone https://github.com/FelipeCamargoXavier/nlw-01.git

# Go into the repository
$ cd nlw-01/server

# Install dependencies
$ npm install

# Run Migrates
$ npm run knex:migrate

# Run Seeds
$ npm run knex:seed

# Start server
$ npm run dev
  ```

# Install and run Web Front-end

```
# Clone this repository
$ git clone https://github.com/FelipeCamargoXavier/nlw-01.git

# Go into the repository
$ cd nlw-01/web

# Install dependencies
$ npm install / yarn install

# Start App
$ npm start
```
